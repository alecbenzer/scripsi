+ (Array) score(str)

generate a ‘score’ for a string used for storing it in a sorted set

def self.score(str)
  str = str.downcase
  scrs = []
  str.split('').each_slice(partition_size) do |s|
    mult = 1.0
    scr = 0.0
    s.each do |char|
      mult /= 27
      scr += (char.ord-'a'.ord+1)*mult if ('a'..'z').include? char
    end
    scrs << scr
  end
  scrs
end
